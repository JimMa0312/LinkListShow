public static int InsertList(int i,datatype x){
    LinkNode p,s;
    p = GetList(i-1);
    if(p == NULL){
        System.out.println("参数i错误");
        return  0;
    }
    else
    {
        s = new LinkNode();
        s.data = x;
        s.next = p.next;
        p.next = s;
        return 1;
    }
}

public static LinkNode GetList(int i )
{
    LinkNode p = Head;
    int j = 0;
    while(p.next!=NULL && j<i)
    {
        p = p.next ;
        j++;
    }
    if(j == i)
        return p;
    else
        return NULL;
}
